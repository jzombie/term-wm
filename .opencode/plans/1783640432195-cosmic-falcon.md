### 1. New: `crates/term-wm-core/src/notification.rs`

Use `Arc<str>` for message — `clone()` on hot path is just atomic refcount increment:

```rust
use std::sync::Arc;

#[derive(Debug, Clone)]
pub struct Notification {
    pub id: u64,
    pub message: Arc<str>,  // Arc<str> → clone is O(1) refcount bump
}

#[derive(Debug, Default)]
pub struct NotificationQueue {
    notifications: VecDeque<Notification>,
    next_id: u64,
    max_capacity: usize,  // 5
}

impl NotificationQueue {
    pub fn push(&mut self, message: impl Into<String>) -> u64 {
        let id = self.next_id;
        self.next_id += 1;
        // Convert String → Arc<str> once at push time
        let msg: Arc<str> = message.into().into();
        if self.notifications.len() >= self.max_capacity {
            self.notifications.pop_front();
        }
        self.notifications.push_back(Notification { id, message: msg });
        id
    }
    pub fn dismiss(&mut self, id: u64) -> bool;
    pub fn renderable(&self) -> impl DoubleEndedIterator<Item = &Notification>;
    pub fn len(&self) -> usize;
}
```

### 6. Modify: `crates/term-wm-core/src/engine.rs`

In `generate_regions`, after window regions:

```rust
use textwrap::Options;

const TOAST_W: u16 = 40;
const MARGIN: u16 = 2;
const GAP: u16 = 1;

let managed = wm.managed_area();

// Geometric circuit breaker — guard against zero-width inner_w
if managed.width <= MARGIN.saturating_mul(2).saturating_add(2) {
    // terminal too narrow for any toast; skip notification regions
    return;
}

let actual_w = TOAST_W.min(managed.width.saturating_sub(MARGIN.saturating_mul(2)));
let inner_w = actual_w.saturating_sub(2) as usize;
// textwrap options mirror Ratatui's Wrap { trim: true }
let wrap_opts = Options::new(inner_w).trim(true);

let mut y_offset = MARGIN;

for notification in wm.notifications().renderable().rev() {  // newest first
    let lines = textwrap::wrap(&notification.message, &wrap_opts);
    let h = (lines.len() as u16).saturating_add(2);
    let h = h.min(managed.height.saturating_sub(y_offset.saturating_add(MARGIN)));
    if h < 3 { break; }

    let x = managed.width.saturating_sub(actual_w).saturating_sub(MARGIN) as i32;

    plan.push(RenderRegion {
        key: WindowKey::default(),
        bounds: LayoutRect { x, y: y_offset as i32, width: actual_w, height: h },
        z_index: 20,
        dimmed: false,
        region_type: RegionType::Notification(Arc::clone(&notification.message)),
        //        ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
        //        O(1) — atomic refcount increment, no heap alloc
    });

    y_offset = y_offset.saturating_add(h).saturating_add(GAP);
}
```
