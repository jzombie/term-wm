### 6. Modify: `crates/term-wm-core/src/engine.rs`

In `generate_regions`, after window regions, append notification regions using an **accumulating Y offset**:

```rust
use textwrap;

let mut y_offset: u16 = 2;  // MARGIN from top
const TOAST_W: u16 = 40;
const GAP: u16 = 1;

for notification in wm.notifications().renderable() {
    let inner_w = (TOAST_W.saturating_sub(2)) as usize;
    let lines = textwrap::wrap(&notification.message, inner_w);
    let h = (lines.len() as u16).saturating_add(2);  // +borders
    let h = h.min(wm.managed_area().height.saturating_sub(y_offset + 2));
    if h < 3 { break; }

    let x = wm.managed_area().width
        .saturating_sub(TOAST_W)
        .saturating_sub(2) as i32;  // x margin

    plan.push(RenderRegion {
        key: WindowKey::default(),
        bounds: LayoutRect {
            x,
            y: y_offset as i32,
            width: TOAST_W,
            height: h,
        },
        z_index: 20,                  // above overlays (10), below cursor
        dimmed: false,
        region_type: RegionType::Notification(notification.id),
    });

    y_offset = y_offset.saturating_add(h).saturating_add(GAP);
}
```

### 11. Modify: `crates/term-wm-console/src/draw_plan_renderer.rs`

In the render loop, route by `region_type`:

```rust
for region in draw_plan.regions() {
    let area = layout_rect_to_rect(region.bounds);
    match region.region_type {
        RegionType::Window => { /* existing: component_for_key_mut, composite/direct */ }
        RegionType::Notification(id) => {
            // Direct ID lookup — no spatial reverse-engineering
            if let Some(n) = wm.notifications().renderable().find(|n| n.id == id) {
                Clear.render(area, target_buf);
                let block = Block::default()
                    .borders(Borders::ALL)
                    .border_style(Style::default().fg(Color::White))
                    .style(Style::default().bg(Color::DarkGray).fg(Color::White));
                Paragraph::new(n.message.as_str())
                    .block(block)
                    .wrap(Wrap { trim: true })
                    .render(area, target_buf);
            }
        }
    }
}
```

No `textwrap` dependency in `term-wm-console` — Ratatui's `Paragraph::wrap` handles the actual rendering; the pre-calculation in `engine.rs` uses `textwrap` (core crate).
