# Revised Implementation Plan: Floating/Tiling Window Behavior

## Algorithm: Topology-Aware Balanced Insertion
Instead of `split_root` (which bisects the root), the balanced algorithm:
1. Computes current regions from the tree + workspace bounds
2. Finds the leaf with the largest area (`width * height`)
3. Splits it along its longer axis (`Horizontal` if `width >= height`, else `Vertical`)
4. Inserts the new window adjacent via `insert_leaf`

This ensures each window gets proportional space. The Euclidean-sorted ORDER determines which windows get inserted first (and thus get the largest remaining areas).

```rust
pub(super) fn float_all_windows(&mut self) {
    let tiled_keys: Vec<_> = self.regions.ids()
        .iter()
        .filter(|key| !self.is_window_floating(**key))
        .copied()
        .collect();

    if tiled_keys.is_empty() { return; }

    let regions: Vec<_> = tiled_keys.iter()
        .map(|key| (*key, self.region(*key)))
        .collect();

    self.managed_layout = None;

    for (key, rect) in regions {
        self.set_floating_rect(key,
            Some(crate::window::FloatRectSpec::Absolute(
                crate::window::FloatRect {
                    x: rect.x, y: rect.y,
                    width: rect.width.max(1), height: rect.height.max(1),
                },
            )),
        );
    }
}
```

## Step 3: Euclidean sort (takes ownership)
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
fn calculate_tiling_order(
    mut floating_rects: Vec<(WindowKey, Rect)>,
    target_rect: Rect,
) -> Vec<(WindowKey, Rect)> {
    let target_center_y = (target_rect.y as i32) + (target_rect.height as i32) / 2;
    let target_center_x = (target_rect.x as i32) + (target_rect.width as i32) / 2;

    floating_rects.sort_by(|a, b| {
        // Scale Y by 2 to account for terminal cell aspect ratio (~2:1 height:width)
        let a_dy = (((a.1.y as i32) + (a.1.height as i32) / 2) - target_center_y) * 2;
        let a_dx = ((a.1.x as i32) + (a.1.width as i32) / 2) - target_center_x;
        let b_dy = (((b.1.y as i32) + (b.1.height as i32) / 2) - target_center_y) * 2;
        let b_dx = ((b.1.x as i32) + (b.1.width as i32) / 2) - target_center_x;
        (a_dx * a_dx + a_dy * a_dy).cmp(&(b_dx * b_dx + b_dy * b_dy))
    });
    floating_rects
}
```

## Step 4: `TilingLayout::insert_window_balanced()`
**File**: `crates/term-wm-core/src/layout/tiling.rs`

```rust
/// Topology-aware insertion: finds the largest leaf by area,
/// splits along its longer axis, inserts `insert` adjacent.
/// Both preview simulation and commit use this exact method.
pub fn insert_window_balanced(&mut self, insert: Id, area: Rect) {
    let regions = self.layout(area);
    if regions.is_empty() {
        self.split_root(insert, InsertPosition::Right);
        return;
    }

    // Find leaf with largest area
    let (largest_id, largest_rect) = regions.iter()
        .max_by_key(|(_, r)| (r.width as u32) * (r.height as u32))
        .copied()
        .unwrap();

    // Split along the longer axis (scale height by 2 for terminal cell aspect ratio)
    let pos = if largest_rect.width >= largest_rect.height * 2 {
        InsertPosition::Right
    } else {
        InsertPosition::Bottom
    };

    if !self.root.insert_leaf(largest_id, insert, pos) {
        self.split_root(insert, pos);
    }
}
```

```rust
fn simulate_position_based_layout(
    windows: Vec<(WindowKey, Rect)>,
    anchor_key: WindowKey,
    workspace_bounds: Rect,
) -> Vec<(WindowKey, Rect)> {
    if windows.is_empty() { return Vec::new(); }

    let target_rect = windows.iter()
        .find(|(k, _)| *k == anchor_key)
        .map(|(_, r)| *r)
        .expect("anchor_key must be in windows");

    let sorted = calculate_tiling_order(windows, target_rect);
    let sorted_keys: Vec<WindowKey> = sorted.iter().map(|(k, _)| *k).collect();

    let mut layout = TilingLayout::new(LayoutNode::leaf(sorted[0].0));
    for key in &sorted_keys[1..] {
        layout.insert_window_balanced(*key, workspace_bounds);
    }

    // Re-map to preserve sorted key order (BSP traversal may reorder)
    let mut regions = layout.layout(workspace_bounds);
    sorted_keys.iter().filter_map(|k| {
        let idx = regions.iter().position(|(rk, _)| rk == k)?;
        Some((*k, regions.swap_remove(idx).1))
    }).collect()
}
```

```rust
pub(super) fn apply_position_based_layout(&mut self, anchor_key: WindowKey, area: Rect) {
    let windows: Vec<_> = self.regions.ids()
        .iter()
        .filter(|key| self.is_window_floating(**key))
        .map(|key| (*key, self.region(*key)))
        .collect();

    if windows.is_empty() { return; }

    let target_rect = windows.iter()
        .find(|(k, _)| *k == anchor_key)
        .map(|(_, r)| *r)
        .expect("anchor_key must be in windows");

    let sorted = calculate_tiling_order(windows, target_rect);
    let sorted_keys: Vec<WindowKey> = sorted.iter().map(|(k, _)| *k).collect();

    for key in &sorted_keys { self.clear_floating_rect(*key); }

    let mut layout = TilingLayout::new(LayoutNode::leaf(sorted_keys[0]));
    for key in &sorted_keys[1..] {
        layout.insert_window_balanced(*key, area);
    }
    self.managed_layout = Some(layout);
}
```

## Step 6: Snap preview cache
**File**: `crates/term-wm-core/src/window/window_manager/mod.rs`

```rust
struct SnapPreviewCache {
    positions: Vec<(WindowKey, Rect)>,
    cached_rect: Option<Rect>,
    cached_quadrant: Option<(bool, bool)>,
    dragged_key: Option<WindowKey>,
}

impl SnapPreviewCache {
    fn needs_recalc(&self, mouse_x: u16, mouse_y: u16, dragged: WindowKey) -> bool {
        let Some(rect) = self.cached_rect else { return true; };
        if self.dragged_key != Some(dragged) { return true; }
        let out_of_bounds = mouse_x < rect.x
            || mouse_x >= rect.x + rect.width
            || mouse_y < rect.y
            || mouse_y >= rect.y + rect.height;
        let cur_q = (mouse_x >= rect.x + rect.width / 2, mouse_y >= rect.y + rect.height / 2);
        out_of_bounds || self.cached_quadrant != Some(cur_q)
    }

    fn update(&mut self, mouse_x: u16, mouse_y: u16, rect: Rect, dragged: WindowKey, pos: Vec<(WindowKey, Rect)>) {
        self.cached_quadrant = Some((mouse_x >= rect.x + rect.width / 2, mouse_y >= rect.y + rect.height / 2));
        self.cached_rect = Some(rect);
        self.dragged_key = Some(dragged);
        self.positions = pos;
    }

    fn clear(&mut self) {
        self.cached_rect = None;
        self.cached_quadrant = None;
        self.dragged_key = None;
        self.positions.clear();
    }
}
```

Add to `WindowManager`: `snap_preview_cache: SnapPreviewCache`.
