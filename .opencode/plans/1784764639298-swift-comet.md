# Revised Implementation Plan: Floating/Tiling Window Behavior

## Fixes This Round

1. **Anchor is explicit**: `tile_all_windows_position_based(key)` takes the dragged window's key
2. **Read before erase**: `target_rect` extracted from the pre-collected vector, before clearing
3. **Quadrant caching**: Cache invalidates only when mouse exits target window's rect or crosses a quadrant boundary (4+ cell grid), not on every cell move
4. **No redundant detach**: `execute_float_all` skips `detach_to_floating` — uses atomic layout drop directly

## Implementation Plan

### Step 1: `float_all_windows()` — atomic layout drop
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
pub(super) fn float_all_windows(&mut self) {
    let tiled_keys: Vec<_> = self.regions.ids()
        .iter()
        .filter(|key| !self.is_window_floating(**key))
        .copied()
        .collect();

    if tiled_keys.is_empty() {
        return;
    }

    let regions: Vec<_> = tiled_keys.iter()
        .map(|key| (*key, self.region(*key)))
        .collect();

    // Atomic: discard entire BSP tree at once
    self.managed_layout = None;

    for (key, rect) in regions {
        self.set_floating_rect(
            key,
            Some(crate::window::FloatRectSpec::Absolute(
                crate::window::FloatRect {
                    x: rect.x,
                    y: rect.y,
                    width: rect.width.max(1),
                    height: rect.height.max(1),
                },
            )),
        );
    }
}
```

pub(super) fn execute_float_all(&mut self, key: WindowKey) {
    if self.is_window_floating(key) {
        return;
    }
    self.float_all_windows();
    // Ensure the triggering window is included (it was tiled, float_all caught it)
    if !self.is_window_floating(key) {
        let rect = self.region(key);
        self.set_floating_rect(
            key,
            Some(crate::window::FloatRectSpec::Absolute(
                crate::window::FloatRect {
                    x: rect.x,
                    y: rect.y,
                    width: rect.width.max(1),
                    height: rect.height.max(1),
                },
            )),
        );
    }
}

### Step 3: Euclidean spatial sort
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
fn calculate_tiling_order(
    floating_rects: &[(WindowKey, Rect)],
    target_rect: Rect,
) -> Vec<(WindowKey, Rect)> {
    let target_center_y = (target_rect.y as i32) + (target_rect.height as i32) / 2;
    let target_center_x = (target_rect.x as i32) + (target_rect.width as i32) / 2;

    let mut sorted = floating_rects.to_vec();
    sorted.sort_by(|a, b| {
        let a_dy = ((a.1.y as i32) + (a.1.height as i32) / 2) - target_center_y;
        let a_dx = ((a.1.x as i32) + (a.1.width as i32) / 2) - target_center_x;
        let b_dy = ((b.1.y as i32) + (b.1.height as i32) / 2) - target_center_y;
        let b_dx = ((b.1.x as i32) + (b.1.width as i32) / 2) - target_center_x;

        (a_dx * a_dx + a_dy * a_dy).cmp(&(b_dx * b_dx + b_dy * b_dy))
    });
    sorted
}
```

### Step 4: `tile_all_windows_position_based(anchor_key)` — read before erase
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
pub(super) fn tile_all_windows_position_based(&mut self, anchor_key: WindowKey) {
    // Collect all floating rects BEFORE clearing anything
    let floating_rects: Vec<_> = self.regions.ids()
        .iter()
        .filter(|key| self.is_window_floating(**key))
        .map(|key| {
            let rect = self.region(*key);
            (*key, rect)
        })
        .collect();

    if floating_rects.is_empty() {
        return;
    }

    // Extract target rect from the pre-collected data (DO NOT read region after clear)
    let target_rect = floating_rects
        .iter()
        .find(|(k, _)| *k == anchor_key)
        .map(|(_, r)| *r)
        .expect("anchor_key must be in floating rects");

    // Sort by Euclidean distance to the anchor window's center
    let sorted = calculate_tiling_order(&floating_rects, target_rect);

    // Now clear floating state for all
    for (key, _) in &sorted {
        self.clear_floating_rect(*key);
    }

    // Rebuild layout: first window is the anchor
    let first = sorted[0];
    self.managed_layout = Some(TilingLayout::new(LayoutNode::leaf(first.0)));

    for (key, _) in &sorted[1..] {
        self.tile_window_key(*key);
    }
}
```

### Step 5: Quadrant-based snap preview cache
**File**: `crates/term-wm-core/src/window/window_manager/mod.rs`

```rust
struct SnapPreviewCache {
    positions: Vec<(WindowKey, Rect)>,
    /// Cached quadrant region boundaries for invalidation.
    cached_rect: Option<Rect>,
    /// Which quadrant of the rect the cursor was in.
    cached_quadrant: Option<(bool, bool)>,
    dragged_key: Option<WindowKey>,
}

impl SnapPreviewCache {
    fn needs_recalc(&self, mouse_x: u16, mouse_y: u16, dragged: WindowKey) -> bool {
        let Some(rect) = self.cached_rect else {
            return true;
        };
        if self.dragged_key != Some(dragged) {
            return true;
        }
        // Compute current quadrant relative to the target rect
        let mid_x = rect.x + (rect.width as i32) / 2;
        let mid_y = rect.y + (rect.height as i32) / 2;
        let cur_q = (mouse_x as i32 >= mid_x, mouse_y as i32 >= mid_y);
        // Only invalidate when quadrant changes
        self.cached_quadrant != Some(cur_q)
    }

    fn update(&mut self, mouse_x: u16, mouse_y: u16, rect: Rect, dragged: WindowKey, pos: Vec<(WindowKey, Rect)>) {
        let mid_x = rect.x + (rect.width as i32) / 2;
        let mid_y = rect.y + (rect.height as i32) / 2;
        self.cached_quadrant = Some((mouse_x as i32 >= mid_x, mouse_y as i32 >= mid_y));
        self.cached_rect = Some(rect);
        self.dragged_key = Some(dragged);
        self.positions = pos;
    }

    fn clear(&mut self) {
        self.cached_rect = None;
        self.cached_quadrant = None;
        self.dragged_key = None;
        self.positions.clear();
    }
}
```
