### Step 1: Add `float_all_windows()` method
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
pub(super) fn float_all_windows(&mut self) {
    let tiled_keys: Vec<_> = self.regions.ids()
        .iter()
        .filter(|key| !self.is_window_floating(**key))
        .copied()
        .collect();

    if tiled_keys.is_empty() {
        return;
    }

    let regions: Vec<_> = tiled_keys.iter()
        .map(|key| (*key, self.region(*key)))
        .collect();

    self.managed_layout = None;

    for (key, rect) in regions {
        self.set_floating_rect(
            key,
            Some(crate::window::FloatRectSpec::Absolute(
                crate::window::FloatRect {
                    x: rect.x,
                    y: rect.y,
                    width: rect.width.max(1),
                    height: rect.height.max(1),
                },
            )),
        );
    }
}
```

pub(super) fn execute_float_all(&mut self, key: WindowKey) {
    if self.is_window_floating(key) {
        return;
    }

    let rect = self.region(key);
    self.detach_to_floating(key, rect);
    self.float_all_windows();
}

### Step 3: Add Euclidean spatial sort
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
fn calculate_tiling_order(
    floating_rects: &[(WindowKey, Rect)],
    target_rect: Rect,
) -> Vec<(WindowKey, Rect)> {
    let target_center_y = (target_rect.y as i32) + (target_rect.height as i32) / 2;
    let target_center_x = (target_rect.x as i32) + (target_rect.width as i32) / 2;

    let mut sorted = floating_rects.to_vec();
    sorted.sort_by(|a, b| {
        let a_dy = ((a.1.y as i32) + (a.1.height as i32) / 2) - target_center_y;
        let a_dx = ((a.1.x as i32) + (a.1.width as i32) / 2) - target_center_x;
        let b_dy = ((b.1.y as i32) + (b.1.height as i32) / 2) - target_center_y;
        let b_dx = ((b.1.x as i32) + (b.1.width as i32) / 2) - target_center_x;

        (a_dx * a_dx + a_dy * a_dy).cmp(&(b_dx * b_dx + b_dy * b_dy))
    });
    sorted
}
```

### Step 4: Add `tile_all_windows_position_based()` (uses the sort)
**File**: `crates/term-wm-core/src/window/window_manager/drag.rs`

```rust
pub(super) fn tile_all_windows_position_based(&mut self) {
    let floating_rects: Vec<_> = self.regions.ids()
        .iter()
        .filter(|key| self.is_window_floating(**key))
        .map(|key| {
            let rect = self.region(*key);
            (*key, rect)
        })
        .collect();

    if floating_rects.is_empty() {
        return;
    }

    for (key, _) in &floating_rects {
        self.clear_floating_rect(*key);
    }

    let anchor_key = floating_rects[0].0;
    let target_rect = self.region(anchor_key);
    let sorted = calculate_tiling_order(&floating_rects, target_rect);
    let first = sorted[0];

    self.managed_layout = Some(crate::layout::TilingLayout::new(
        crate::layout::LayoutNode::leaf(first.0),
    ));

    for (key, _) in &sorted[1..] {
        self.tile_window_key(*key);
    }
}
```

### Step 5: Snap preview cache with invalidation
**File**: `crates/term-wm-core/src/window/window_manager/mod.rs`

```rust
struct SnapPreviewCache {
    positions: Vec<(WindowKey, Rect)>,
    /// Last mouse cell that triggered recalculation.
    last_mouse: Option<(u16, u16)>,
    /// Which window we were dragging when last calculated.
    dragged_key: Option<WindowKey>,
}

impl SnapPreviewCache {
    fn needs_recalc(&self, mouse_x: u16, mouse_y: u16, dragged: WindowKey) -> bool {
        match self.last_mouse {
            Some((lx, ly)) => {
                self.dragged_key != Some(dragged)
                    || mouse_x.abs_diff(lx) >= 1
                    || mouse_y.abs_diff(ly) >= 1
            }
            None => true,
        }
    }

    fn update(&mut self, mouse_x: u16, mouse_y: u16, dragged: WindowKey, pos: Vec<(WindowKey, Rect)>) {
        self.last_mouse = Some((mouse_x, mouse_y));
        self.dragged_key = Some(dragged);
        self.positions = pos;
    }

    fn clear(&mut self) {
        self.last_mouse = None;
        self.dragged_key = None;
        self.positions.clear();
    }
}
```
