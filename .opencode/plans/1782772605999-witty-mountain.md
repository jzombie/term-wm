# Plan: End-to-end panic → debug log test

## Approach

Call `run_app()` with a real `WindowManager`, real `EventSource`, real `RenderTarget` (wrapping `ratatui::TestBackend`). The draw callback panics on the first call. After the panic hook fires and the debug window opens, the callback closes it so the quit condition triggers. Then verify the `DebugLogHandle` contains the panic message and backtrace.

## Files to modify

### `crates/term-wm-sys-ui-components/src/wm_debug_log.rs`

Add `pub fn lines()` to `DebugLogHandle`:

```rust
impl DebugLogHandle {
    pub fn lines(&self) -> Vec<String> {
        self.inner.lock()
            .map(|buf| buf.lines.iter().cloned().collect())
            .unwrap_or_default()
    }
}
```

### `tests/panic_debug_log.rs` (NEW)

```rust
use std::io;
use std::sync::atomic::{AtomicUsize, Ordering};
use std::sync::Arc;
use std::time::Duration;

use crossterm::event::{Event, KeyEvent, MouseEvent};
use ratatui::backend::TestBackend;
use ratatui::prelude::Rect;
use ratatui::Terminal;

use term_wm::io::{EventSource, PowerProfile, RenderTarget};
use term_wm::ui::UiFrame;
use term_wm::window::{WindowDrawContext, WindowId, WindowManager, SystemWindowId, WmMenuAction};
use term_wm::window::decorator::WindowDecoratorOverride;
use term_wm::wm_config::WmConfig;
use term_wm::app_context::AppContext;
use term_wm::runner::{WindowManagerHost, WindowProvider, run_app};
use term_wm::MenuOverlay;

// Minimal test render target backed by TestBackend
#[derive(Debug)]
struct TestOutput {
    terminal: Terminal<TestBackend>,
}

impl TestOutput {
    fn new() -> io::Result<Self> {
        let backend = TestBackend::new(80, 24);
        let terminal = Terminal::new(backend)?;
        Ok(Self { terminal })
    }
}

impl RenderTarget for TestOutput {
    type Backend = TestBackend;
    fn enter(&mut self) -> io::Result<()> { Ok(()) }
    fn exit(&mut self) -> io::Result<()> { Ok(()) }
    fn draw<F>(&mut self, f: F) -> io::Result<()>
    where F: FnOnce(UiFrame<'_>) {
        self.terminal.draw(|frame| {
            let wrapper = UiFrame::new(frame);
            f(wrapper);
        }).map_err(|e| io::Error::other(e.to_string()))
    }
}

// Event source that always returns immediately with no events
#[derive(Debug)]
struct ImmediateDriver;
impl EventSource for ImmediateDriver {
    fn poll(&mut self, _timeout: Duration) -> io::Result<bool> { Ok(false) }
    fn read(&mut self) -> io::Result<Event> {
        Err(io::Error::other("poll never returns true"))
    }
    fn next_key(&mut self) -> io::Result<KeyEvent> {
        Err(io::Error::other("not used"))
    }
    fn next_mouse(&mut self) -> io::Result<MouseEvent> {
        Err(io::Error::other("not used"))
    }
}

// Minimal app with no windows
struct SparseApp {
    wm: WindowManager<usize>,
}
impl WindowManagerHost<usize> for SparseApp {
    fn windows(&mut self) -> &mut WindowManager<usize> { &mut self.wm }
}
impl WindowProvider<usize> for SparseApp {
    fn enumerate_windows(&mut self) -> Vec<usize> { vec![] }
    fn render_window(&mut self, _frame: &mut UiFrame<'_>, _window: WindowDrawContext<usize>) {}
}

#[test]
fn render_panic_shows_in_debug_log() {
    // Wire up the real panic hook + global log handle
    let (_comp, handle) =
        term_wm_sys_ui_components::WmDebugLogComponent::new(2000);
    assert!(
        term_wm_sys_ui_components::set_global_debug_log(handle.clone()),
        "set_global_debug_log should succeed (first call)"
    );
    term_wm_sys_ui_components::install_panic_hook();

    // Build the WindowManager and app
    let menu: Box<dyn MenuOverlay<WmMenuAction>> =
        Box::new(term_wm_sys_ui_components::WmMenuOverlay::new());
    let wm = WindowManager::<usize>::with_config(
        0,
        WmConfig::standalone(),
        Arc::new(AppContext::new("test", "0.0.0")),
        None::<Box<dyn term_wm::top_panel_trait::TopPanel<_>>>,
        None::<Box<dyn term_wm::BottomPanel>>,
        menu,
    );

    let mut app = SparseApp { wm };
    let mut output = TestOutput::new().expect("TestOutput creation");
    let mut driver = ImmediateDriver;
    let focus_regions: Vec<usize> = vec![];

    let panic_msg = "intentional-panic-from-draw";
    let draw_count = Arc::new(AtomicUsize::new(0));

    let result = run_app(
        &mut output,
        &mut driver,
        &mut app,
        &focus_regions,
        |id| id,
        {
            let draw_count = draw_count.clone();
            move |_frame, app| {
                let call = draw_count.fetch_add(1, Ordering::SeqCst);
                if call == 0 {
                    panic!("{}", panic_msg);
                } else {
                    // Close the debug window so the idle-path
                    // quit condition (no windows + no system
                    // windows) fires.
                    let _ = app.windows().close_window(
                        WindowId::System(SystemWindowId::DebugLog),
                    );
                }
            }
        },
    );

    assert!(result.is_ok(), "run_app should return Ok after panic");

    let lines = handle.lines();
    let joined = lines.join("\n");
    assert!(
        joined.contains(panic_msg),
        "panic message should appear in debug log\n=== log ===\n{joined}\n=========="
    );
    assert!(
        joined.contains("stack backtrace"),
        "backtrace should appear in debug log\n=== log ===\n{joined}\n=========="
    );
}
```
